## Quest 1. sort.c (динамический массив)

### Задание

Модифицировать сортировку из D07T04 для работы с динамическим массивом (malloc/calloc).

### Полный код

```c
// src/sort.c

#include <stdio.h>
#include <stdlib.h>

int main(void) {
    int n;
    if (scanf("%d", &n) != 1 || n <= 0) {
        printf("n/a");
        return 0;
    }

    // Динамическое выделение памяти
    int *a = (int *)malloc(n * sizeof(int));
    if (a == NULL) {
        printf("n/a");
        return 0;
    }

    // Ввод
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    // Сортировка выбором
    for (int i = 0; i < n - 1; i++) {
        int min_idx = i;
        for (int j = i + 1; j < n; j++) {
            if (a[j] < a[min_idx]) min_idx = j;
        }
        int temp = a[i];
        a[i] = a[min_idx];
        a[min_idx] = temp;
    }

    // Вывод
    for (int i = 0; i < n; i++) {
        if (i > 0) printf(" ");
        printf("%d", a[i]);
    }

    // Освобождение памяти
    free(a);

    return 0;
}
```

### Разбор: malloc

```c
int *a = (int *)malloc(n * sizeof(int));
```

| Часть | Что делает |
|-------|-----------|
| `malloc(n * sizeof(int))` | Выделяет n * 4 байта (размер int) |
| `(int *)` | Приведение типа (void* → int*) |
| `sizeof(int)` | Размер одного int в байтах (обычно 4) |
| `NULL` | Если память не выделена |

### Разбор: free

```c
free(a);  // ОБЯЗАТЕЛЬНО! Иначе — утечка памяти
```

## Quest 3. matrix.c (4 способа выделения)

### Задание

Реализовать ввод/вывод матриц с 4 способами выделения памяти:
1. Статический (max 100×100)
2. Динамический массив указателей
3. Динамический массив указателей на сегменты одного буфера
4. Динамический массив указателей на сегменты с calloc

### Полный код

```c
// src/matrix.c

#include <stdio.h>
#include <stdlib.h>

// Способ 1: Статический
void static_alloc(int rows, int cols) {
    int matrix[100][100];
    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            scanf("%d", &matrix[i][j]);

    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            if (j > 0) printf(" ");
            printf("%d", matrix[i][j]);
        }
        if (i < rows - 1) printf("\n");
    }
}

// Способ 2: Динамический массив указателей
void dynamic_pointers(int rows, int cols) {
    int **matrix = (int **)malloc(rows * sizeof(int *));
    for (int i = 0; i < rows; i++)
        matrix[i] = (int *)malloc(cols * sizeof(int));

    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            scanf("%d", &matrix[i][j]);

    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            if (j > 0) printf(" ");
            printf("%d", matrix[i][j]);
        }
        if (i < rows - 1) printf("\n");
    }

    for (int i = 0; i < rows; i++) free(matrix[i]);
    free(matrix);
}

// Способ 3: Один буфер + указатели на сегменты
void single_buffer(int rows, int cols) {
    int *buffer = (int *)malloc(rows * cols * sizeof(int));
    int **matrix = (int **)malloc(rows * sizeof(int *));
    for (int i = 0; i < rows; i++)
        matrix[i] = buffer + i * cols;

    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            scanf("%d", &matrix[i][j]);

    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            if (j > 0) printf(" ");
            printf("%d", matrix[i][j]);
        }
        if (i < rows - 1) printf("\n");
    }

    free(matrix);
    free(buffer);
}

// Способ 4: calloc (инициализация нулями)
void calloc_alloc(int rows, int cols) {
    int **matrix = (int **)calloc(rows, sizeof(int *));
    for (int i = 0; i < rows; i++)
        matrix[i] = (int *)calloc(cols, sizeof(int));

    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            scanf("%d", &matrix[i][j]);

    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            if (j > 0) printf(" ");
            printf("%d", matrix[i][j]);
        }
        if (i < rows - 1) printf("\n");
    }

    for (int i = 0; i < rows; i++) free(matrix[i]);
    free(matrix);
}

int main(void) {
    int method;
    int rows, cols;

    if (scanf("%d", &method) != 1) { printf("n/a"); return 0; }
    if (scanf("%d %d", &rows, &cols) != 2) { printf("n/a"); return 0; }

    if (method == 1) static_alloc(rows, cols);
    else if (method == 2) dynamic_pointers(rows, cols);
    else if (method == 3) single_buffer(rows, cols);
    else if (method == 4) calloc_alloc(rows, cols);
    else { printf("n/a"); return 0; }

    return 0;
}
```

### Разбор: 4 способа

| Способ | Память | Освобождение |
|--------|--------|--------------|
| Статический | Стек (автоматически) | Не нужно |
| malloc указателей | Куча (отдельно на строку) | `free(matrix[i])` для каждой + `free(matrix)` |
| Один буфер | Куча (один блок) | `free(buffer)` + `free(matrix)` |
| calloc | Куча (нули + отдельно) | `free(matrix[i])` для каждой + `free(matrix)` |

### malloc vs calloc

```c
malloc(n, size)  // Выделяет память, НЕ обнуляет
calloc(n, size)  // Выделяет память и ЗАПОЛНЯЕТ НУЛЯМИ
```

## Quest 4. matrix_extended.c (мин/макс)

### Задание

Расширить matrix.c: после вывода матрицы показать максимум в каждой строке и минимум в каждом столбце.

### Полный код

```c
// src/matrix_extended.c

#include <stdio.h>
#include <stdlib.h>

void process(int method, int rows, int cols) {
    int **matrix = (int **)malloc(rows * sizeof(int *));
    int *buffer = NULL;

    if (method == 1) {
        // Статический (max 100x100)
        static int static_mat[100][100];
        for (int i = 0; i < rows; i++)
            for (int j = 0; j < cols; j++)
                scanf("%d", &static_mat[i][j]);

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (j > 0) printf(" ");
                printf("%d", static_mat[i][j]);
            }
            if (i < rows - 1) printf("\n");
        }
        printf("\n");

        // Макс по строкам
        for (int i = 0; i < rows; i++) {
            int mx = static_mat[i][0];
            for (int j = 1; j < cols; j++)
                if (static_mat[i][j] > mx) mx = static_mat[i][j];
            if (i > 0) printf(" ");
            printf("%d", mx);
        }
        printf("\n");

        // Мин по столбцам
        for (int j = 0; j < cols; j++) {
            int mn = static_mat[0][j];
            for (int i = 1; i < rows; i++)
                if (static_mat[i][j] < mn) mn = static_mat[i][j];
            if (j > 0) printf(" ");
            printf("%d", mn);
        }
        return;
    }

    // Динамические способы
    buffer = (int *)malloc(rows * cols * sizeof(int));
    for (int i = 0; i < rows; i++)
        matrix[i] = buffer + i * cols;

    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            scanf("%d", &matrix[i][j]);

    // Вывод матрицы
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            if (j > 0) printf(" ");
            printf("%d", matrix[i][j]);
        }
        if (i < rows - 1) printf("\n");
    }
    printf("\n");

    // Макс по строкам
    for (int i = 0; i < rows; i++) {
        int mx = matrix[i][0];
        for (int j = 1; j < cols; j++)
            if (matrix[i][j] > mx) mx = matrix[i][j];
        if (i > 0) printf(" ");
        printf("%d", mx);
    }
    printf("\n");

    // Мин по столбцам
    for (int j = 0; j < cols; j++) {
        int mn = matrix[0][j];
        for (int i = 1; i < rows; i++)
            if (matrix[i][j] < mn) mn = matrix[i][j];
        if (j > 0) printf(" ");
        printf("%d", mn);
    }

    free(matrix);
    free(buffer);
}

int main(void) {
    int method, rows, cols;
    scanf("%d %d %d", &method, &rows, &cols);
    process(method, rows, cols);
    return 0;
}
```

## Quest 6. matrix_arithmetic.c (арифметика)

### Задание

Реализовать операции над матрицами: 1 — сумма, 2 — умножение, 3 — транспонирование.

### Полный код

```c
// src/matrix_arithmetic.c

#include <stdio.h>
#include <stdlib.h>

void input(int **m, int rows, int cols) {
    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            scanf("%d", &m[i][j]);
}

void output(int **m, int rows, int cols) {
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            if (j > 0) printf(" ");
            printf("%d", m[i][j]);
        }
        if (i < rows - 1) printf("\n");
    }
}

void sum(int **a, int **b, int **result, int rows, int cols) {
    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            result[i][j] = a[i][j] + b[i][j];
}

void mul(int **a, int **b, int **result, int r1, int c1, int c2) {
    for (int i = 0; i < r1; i++)
        for (int j = 0; j < c2; j++) {
            result[i][j] = 0;
            for (int k = 0; k < c1; k++)
                result[i][j] += a[i][k] * b[k][j];
        }
}

void transpose(int **a, int **result, int rows, int cols) {
    for (int i = 0; i < rows; i++)
        for (int j = 0; j < cols; j++)
            result[j][i] = a[i][j];
}

int main(void) {
    int op;
    scanf("%d", &op);

    if (op == 1) {
        int rows, cols;
        scanf("%d %d", &rows, &cols);

        int **a = (int **)malloc(rows * sizeof(int *));
        int **b = (int **)malloc(rows * sizeof(int *));
        int **r = (int **)malloc(rows * sizeof(int *));
        int *abuf = (int *)malloc(rows * cols * sizeof(int));
        int *bbuf = (int *)malloc(rows * cols * sizeof(int));
        int *rbuf = (int *)malloc(rows * cols * sizeof(int));
        for (int i = 0; i < rows; i++) {
            a[i] = abuf + i * cols;
            b[i] = bbuf + i * cols;
            r[i] = rbuf + i * cols;
        }

        input(a, rows, cols);
        input(b, rows, cols);
        sum(a, b, r, rows, cols);
        output(r, rows, cols);

        free(rbuf); free(bbuf); free(abuf);
        free(r); free(b); free(a);

    } else if (op == 2) {
        int r1, c1, r2, c2;
        scanf("%d %d", &r1, &c1);
        int **a = (int **)malloc(r1 * sizeof(int *));
        int *abuf = (int *)malloc(r1 * c1 * sizeof(int));
        for (int i = 0; i < r1; i++) a[i] = abuf + i * c1;
        input(a, r1, c1);

        scanf("%d %d", &r2, &c2);
        int **b = (int **)malloc(r2 * sizeof(int *));
        int *bbuf = (int *)malloc(r2 * c2 * sizeof(int));
        for (int i = 0; i < r2; i++) b[i] = bbuf + i * c2;
        input(b, r2, c2);

        if (c1 != r2) { printf("n/a"); free(abuf); free(bbuf); free(a); free(b); return 0; }

        int **r = (int **)malloc(r1 * sizeof(int *));
        int *rbuf = (int *)malloc(r1 * c2 * sizeof(int));
        for (int i = 0; i < r1; i++) r[i] = rbuf + i * c2;

        mul(a, b, r, r1, c1, c2);
        output(r, r1, c2);

        free(rbuf); free(bbuf); free(abuf);
        free(r); free(b); free(a);

    } else if (op == 3) {
        int rows, cols;
        scanf("%d %d", &rows, &cols);
        int **a = (int **)malloc(rows * sizeof(int *));
        int *abuf = (int *)malloc(rows * cols * sizeof(int));
        for (int i = 0; i < rows; i++) a[i] = abuf + i * cols;
        input(a, rows, cols);

        int **r = (int **)malloc(cols * sizeof(int *));
        int *rbuf = (int *)malloc(cols * rows * sizeof(int));
        for (int i = 0; i < cols; i++) r[i] = rbuf + i * rows;

        transpose(a, r, rows, cols);
        output(r, cols, rows);

        free(rbuf); free(abuf);
        free(r); free(a);
    }

    return 0;
}
```

### Разбор: умножение матриц

```
A (2×3) × B (3×2) = C (2×2)

C[i][j] = Σ A[i][k] * B[k][j]  (k от 0 до c1-1)
```

### Компиляция

## Bonus Quest 8. det.c (определитель)

### Задание

Вычислить определитель квадратной матрицы действительных чисел методом разложения Лапласа.

### Полный код

```c
// src/det.c

#include <math.h>
#include <stdio.h>
#include <stdlib.h>

double det(double **matrix, int n) {
    if (n == 1) return matrix[0][0];
    if (n == 2) return matrix[0][0] * matrix[1][1] - matrix[0][1] * matrix[1][0];

    double result = 0;
    for (int col = 0; col < n; col++) {
        // Создаём минор
        double **minor = (double **)malloc((n - 1) * sizeof(double *));
        double *buf = (double *)malloc((n - 1) * (n - 1) * sizeof(double));
        for (int i = 0; i < n - 1; i++)
            minor[i] = buf + i * (n - 1);

        for (int i = 1; i < n; i++) {
            int mi = 0;
            for (int j = 0; j < n; j++) {
                if (j != col) {
                    minor[i - 1][mi] = matrix[i][j];
                    mi++;
                }
            }
        }

        double sign = (col % 2 == 0) ? 1.0 : -1.0;
        result += sign * matrix[0][col] * det(minor, n - 1);

        free(buf);
        free(minor);
    }
    return result;
}

int main(void) {
    int n;
    scanf("%d", &n);

    double **matrix = (double **)malloc(n * sizeof(double *));
    double *buf = (double *)malloc(n * n * sizeof(double));
    for (int i = 0; i < n; i++)
        matrix[i] = buf + i * n;

    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            scanf("%lf", &matrix[i][j]);

    printf("%.6f", det(matrix, n));

    free(buf);
    free(matrix);
    return 0;
}
```

### Разбор: разложение Лапласа

```
det(A) = Σ (-1)^(i+j) * a[0][j] * det(M[0][j])

Где M[0][j] — минор (матрица без строки 0 и столбца j)
```

```bash

echo "3 3 1 2 3 4 5 6 7 8 9" | ./det
# Вывод: 0.000000
```

## Bonus Quest 9. invert.c (обратная матрица)

### Задание

Вычислить обратную матрицу: A⁻¹ = (1/det(A)) × adj(A)

### Полный код

```c
// src/invert.c

#include <math.h>
#include <stdio.h>
#include <stdlib.h>

double det(double **m, int n) {
    if (n == 1) return m[0][0];
    if (n == 2) return m[0][0]*m[1][1] - m[0][1]*m[1][0];
    double r = 0;
    for (int j = 0; j < n; j++) {
        double **min = (double **)malloc((n-1)*sizeof(double *));
        double *b = (double *)malloc((n-1)*(n-1)*sizeof(double));
        for (int i = 0; i < n-1; i++) min[i] = b + i*(n-1);
        for (int i = 1; i < n; i++) {
            int mi = 0;
            for (int k = 0; k < n; k++) {
                if (k != j) { min[i-1][mi] = m[i][k]; mi++; }
            }
        }
        r += ((j%2==0)?1:-1) * m[0][j] * det(min, n-1);
        free(b); free(min);
    }
    return r;
}

void minor_matrix(double **m, double **result, int n) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            double **min = (double **)malloc((n-1)*sizeof(double *));
            double *b = (double *)malloc((n-1)*(n-1)*sizeof(double));
            for (int k = 0; k < n-1; k++) min[k] = b + k*(n-1);
            int mi = 0;
            for (int r = 0; r < n; r++) {
                if (r == i) continue;
                int mj = 0;
                for (int c = 0; c < n; c++) {
                    if (c == j) continue;
                    min[mi][mj] = m[r][c];
                    mj++;
                }
                mi++;
            }
            double sign = ((i+j)%2==0) ? 1.0 : -1.0;
            result[i][j] = sign * det(min, n-1);
            free(b); free(min);
        }
    }
}

void transpose(double **m, double **result, int n) {
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            result[i][j] = m[j][i];
}

int main(void) {
    int n;
    scanf("%d", &n);

    double **m = (double **)malloc(n*sizeof(double *));
    double *mb = (double *)malloc(n*n*sizeof(double));
    for (int i = 0; i < n; i++) m[i] = mb + i*n;

    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            scanf("%lf", &m[i][j]);

    double d = det(m, n);
    if (fabs(d) < 1e-10) { printf("n/a"); free(mb); free(m); return 0; }

    double **cof = (double **)malloc(n*sizeof(double *));
    double *cofb = (double *)malloc(n*n*sizeof(double));
    for (int i = 0; i < n; i++) cof[i] = cofb + i*n;

    double **adj = (double **)malloc(n*sizeof(double *));
    double *adjb = (double *)malloc(n*n*sizeof(double));
    for (int i = 0; i < n; i++) adj[i] = adjb + i*n;

    minor_matrix(m, cof, n);
    transpose(cof, adj, n);

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (j > 0) printf(" ");
            printf("%.6f", adj[i][j] / d);
        }
        if (i < n - 1) printf("\n");
    }

    free(adjb); free(adj); free(cofb); free(cof); free(mb); free(m);
    return 0;
}
```

### Разбор: обратная матрица

```
A⁻¹ = (1/det(A)) × adj(A)

adj(A) = транспонированная матрица коэффициентов
Коэффициент C[i][j] = (-1)^(i+j) × det(M[i][j])
```
