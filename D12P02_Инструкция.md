## Правила игры Конвея

### Основные правила

Игра "Жизнь" — клеточный автомат на бесконечной двумерной решётке. Каждая клетка может быть **живой** или **мёртвой**.

Соседи клетки — 8 окружающих клеток (по горизонтали, вертикали и диагоналям).

На каждом шаге (тика) правила применяются **одновременно** ко всем клеткам:

| Состояние | Кол-во соседей | Результат |
|-----------|---------------|-----------|
| Живая | < 2 | Умирает (одиночество) |
| Живая | 2 или 3 | Выживает |
| Живая | > 3 | Умирает (перенаселение) |
| Мёртвая | = 3 | Оживает (рождение) |

### Тороидальная решётка

Поле **самозамкнутое** (тор):
- Правый край → Левый край
- Нижний край → Верхний край
- Правый нижний угол → Левый верхний угол

### Параметры проекта

- Размер поля: **80×25** символов
- Живая клетка: `*` (звёздочка)
- Мёртвая клетка: ` ` (пробел)
- Управление: **A/Z** — скорость, **Пробел** — выход

## Полный код game_of_life.c

```c
// src/game_of_life.c

#include <ncurses.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

#define WIDTH 80
#define HEIGHT 25
#define INITIAL_SPEED 200000  // 0.2 секунды

// Прочитать поле из файла
void read_field(int field[HEIGHT][WIDTH], FILE *file) {
    // Обнуляем поле
    for (int i = 0; i < HEIGHT; i++)
        for (int j = 0; j < WIDTH; j++)
            field[i][j] = 0;

    char line[WIDTH + 2];
    int row = 0;
    while (fgets(line, sizeof(line), file) && row < HEIGHT) {
        for (int col = 0; col < WIDTH && line[col] != '\0'; col++) {
            if (line[col] == '*') {
                field[row][col] = 1;
            }
        }
        row++;
    }
}

// Посчитать соседей с учётом тороидальности
int count_neighbors(int field[HEIGHT][WIDTH], int x, int y) {
    int count = 0;
    for (int dx = -1; dx <= 1; dx++) {
        for (int dy = -1; dy <= 1; dy++) {
            if (dx == 0 && dy == 0) continue;  // Пропускаем саму клетку

            int nx = (x + dx + HEIGHT) % HEIGHT;  // Тороидальность по Y
            int ny = (y + dy + WIDTH) % WIDTH;     // Тороидальность по X

            count += field[nx][ny];
        }
    }
    return count;
}

// Один шаг симуляции
void next_generation(int field[HEIGHT][WIDTH]) {
    int new_field[HEIGHT][WIDTH];

    for (int i = 0; i < HEIGHT; i++) {
        for (int j = 0; j < WIDTH; j++) {
            int neighbors = count_neighbors(field, i, j);
            int alive = field[i][j];

            if (alive && (neighbors == 2 || neighbors == 3)) {
                new_field[i][j] = 1;  // Выживает
            } else if (!alive && neighbors == 3) {
                new_field[i][j] = 1;  // Оживает
            } else {
                new_field[i][j] = 0;  // Умирает или остаётся мёртвой
            }
        }
    }

    // Копируем новое поле в старое
    for (int i = 0; i < HEIGHT; i++)
        for (int j = 0; j < WIDTH; j++)
            field[i][j] = new_field[i][j];
}

// Отрисовка поля в ncurses
void draw_field(int field[HEIGHT][WIDTH], int speed) {
    erase();

    for (int i = 0; i < HEIGHT; i++) {
        for (int j = 0; j < WIDTH; j++) {
            if (field[i][j]) {
                mvaddch(i, j, '*');
            }
        }
    }

    // Подсказка внизу
    mvprintw(HEIGHT, 0, "A/Z: speed | Space: pause/quit | Speed: %dms",
             speed / 1000);

    refresh();
}

int main(int argc, char *argv[]) {
    int field[HEIGHT][WIDTH];

    // Чтение начального состояния
    FILE *file = NULL;
    if (argc > 1) {
        file = fopen(argv[1], "r");
    }
    if (file == NULL) {
        file = stdin;  // Читаем из stdin
    }

    read_field(field, file);

    if (file != stdin) {
        fclose(file);
    }

    // Инициализация ncurses
    initscr();
    cbreak();
    noecho();
    curs_set(0);
    keypad(stdscr, TRUE);
    nodelay(stdscr, TRUE);

    int speed = INITIAL_SPEED;
    int ch;

    // Основной цикл
    while (1) {
        draw_field(field, speed);

        // Ввод
        ch = getch();

        if (ch == 'a' || ch == 'A') {
            // Ускорить
            if (speed > 50000) speed -= 50000;
        } else if (ch == 'z' || ch == 'Z') {
            // Замедлить
            if (speed < 1000000) speed += 50000;
        } else if (ch == ' ') {
            break;  // Выход
        }

        // Шаг симуляции
        next_generation(field);
        usleep(speed);
    }

    endwin();
    return 0;
}
```

### Разбор ключевых функций

#### Тороидальная решётка

```c
int nx = (x + dx + HEIGHT) % HEIGHT;
int ny = (y + dy + WIDTH) % WIDTH;
```

`(x + dx + HEIGHT) % HEIGHT` — если вышли за границу, оборачиваемся:
- `x = 0, dx = -1` → `(0 - 1 + 25) % 25 = 24` (нижняя граница)
- `x = 24, dx = 1` → `(24 + 1 + 25) % 25 = 0` (верхняя граница)

#### ncurses функции

| Функция | Что делает |
|---------|-----------|
| `initscr()` | Начало работы с ncurses |
| `endwin()` | Конец работы |
| `erase()` | Очистить экран без мерцания |
| `refresh()` | Обновить реальный экран |
| `mvaddch(y, x, ch)` | Поставить символ в позицию |
| `mvprintw(y, x, fmt, ...)` | Напечатать текст в позиции |
| `getch()` | Прочитать клавишу |
| `nodelay(stdscr, TRUE)` | Неблокирующий ввод |
| `usleep(microsec)` | Задержка |

## Шпаргалка: ncurses

### Основные функции

```c
#include <ncurses.h>

// Инициализация
initscr();            // Начать
endwin();             // Закончить

// Экран
clear();              // Очистить (мерцает)
erase();              // Очистить (без мерцания)
refresh();            // Обновить экран

// Вывод
addch(ch);            // Символ в текущей позиции
mvaddch(y, x, ch);   // Символ в позиции (y, x)
printw(fmt, ...);     // Как printf
mvprintw(y, x, fmt, ...);  // Printf в позиции

// Ввод
ch = getch();         // Прочитать клавишу
nodelay(stdscr, TRUE);  // Неблокирующий getch

// Настройка
cbreak();             // Поимвольный ввод
noecho();             // Не отображать ввод
curs_set(0);          // Скрыть курсор
keypad(stdscr, TRUE); // Спец. клавиши

// Задержка
usleep(microseconds); // Задержка в микросекундах
```

### Спец. клавиши

| Клавиша | Код |
|---------|-----|
| Стрелка вверх | KEY_UP |
| Стрелка вниз | KEY_DOWN |
| Стрелка влево | KEY_LEFT |
| Стрелка вправо | KEY_RIGHT |
| Home | KEY_HOME |
| End | KEY_END |

## Шпаргалка: динамическая память

### malloc / calloc / free

```c
#include <stdlib.h>

// malloc — выделяет память (не обнуляет)
int *arr = (int *)malloc(n * sizeof(int));

// calloc — выделяет и обнуляет
int *arr = (int *)calloc(n, sizeof(int));

// realloc — перераспределяет память
arr = (int *)realloc(arr, new_size * sizeof(int));

// free — освобождает память
free(arr);
arr = NULL;  // Хорошая практика
```

### Типичные ошибки

```c
int *arr = malloc(100);
// Ошибка: забыли приведение типа (в C не обязательно, но в School 21 требуют)
int *arr = (int *)malloc(100);

// Ошибка: не проверили NULL
int *arr = (int *)malloc(100 * sizeof(int));
if (arr == NULL) { /* обработка ошибки */ }

// Ошибка: утечка памяти
int *arr = (int *)malloc(100 * sizeof(int));
// ... используем arr ...
// НЕТ free(arr) → УТЕЧКА!

// Ошибка: двойное освобождение
free(arr);
free(arr);  // НЕЛЬЗЯ!
```
