## Quest 1. maxmin.c

### Задание

Исправить программу поиска максимума и минимума трёх чисел.

### Исходный код (с ошибками)

```c
// src/maxmin.c (ИСХОДНИК С ОШИБКАМИ)
#include <stdio.h>

void maxmin(int a, int b, int c, int max, int min) {
    max = a;
    min = a;
    if (b > max) max = b;
    if (c > max) max = c;
    if (b < min) min = b;
    if (c < min) min = c;
}

int main(void) {
    int a, b, c, max, min;
    scanf("%d%d%d", a, b, c);
    maxmin(a, b, c, max, min);
    printf("%d %d", max, min);
    return 0;
}
```

### Ошибки

1. **`scanf("%d%d%d", a, b, c)`** — нет `&` перед переменными
2. **`void maxmin(..., int max, int min)`** — параметры передаются по значению, изменения не видны в main
3. **`max = b`** — нужно `*max = b` (разыменование указателя)

### Исправленный код

```c
// src/maxmin.c

#include <stdio.h>

void maxmin(int a, int b, int c, int *max, int *min) {
    *max = a;
    *min = a;
    if (b > *max) *max = b;
    if (c > *max) *max = c;
    if (b < *min) *min = b;
    if (c < *min) *min = c;
}

int main(void) {
    int a, b, c, max, min;
    scanf("%d %d %d", &a, &b, &c);
    maxmin(a, b, c, &max, &min);
    printf("%d %d", max, min);
    return 0;
}
```

### Разбор: указатели

| Код | Что делает |
|-----|-----------|
| `int *max` | Параметр — указатель на int (адрес переменной) |
| `&max` | Получить адрес переменной max |
| `*max = a` | Записать значение a по адресу max (разыменование) |
| `*max > b` | Прочитать значение по адресу max и сравнить с b |

### Почему нужен указатель

Функция `maxmin` должна **изменить** значения `max` и `min` в `main`. По значению копия изменяется, оригинал — нет. Указатель передаёт адрес, поэтому функция работает с оригиналом.

### Компиляция

```bash
gcc -Wall -Werror -Wextra -o maxmin src/maxmin.c
echo "3 7 5" | ./maxmin
# Вывод: 7 3
```

#include <stdio.h>

### Исправленный код

```c
// src/squaring.c

#include <stdio.h>

void input(int *a, int n) {
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
}

void squaring(int *a, int n) {
    for (int i = 0; i < n; i++) {
        a[i] = a[i] * a[i];
    }
}

void output(int *a, int n) {
    for (int i = 0; i < n; i++) {
        if (i > 0) printf(" ");
        printf("%d", a[i]);
    }
}

int main(void) {
    int a[10];
    int n;
    scanf("%d", &n);
    input(a, n);
    squaring(a, n);
    output(a, n);
    return 0;
}
```

### Разбор: массивы через указатели

| Код | Что делает |
|-----|-----------|
| `int *a` | Указатель на первый элемент массива |
| `a[i]` | Элемент с индексом i (syntactic sugar для `*(a+i)`) |
| `&a[i]` | Адрес i-го элемента |

### Важно

- `int *a` в параметрах функции — это указатель, но к нему можно обращаться как к массиву: `a[i]`
- Массив **всегда** передаётся в функцию как указатель

## Quest 3. stat.c

### Задание

Реализовать статистический анализ массива: максимум, минимум, среднее, дисперсия.

### Полный код

```c
// src/stat.c

#include <stdio.h>

void input(int *a, int n) {
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
}

void output(int *a, int n) {
    for (int i = 0; i < n; i++) {
        if (i > 0) printf(" ");
        printf("%d", a[i]);
    }
}

int max(int *a, int n) {
    int m = a[0];
    for (int i = 1; i < n; i++) {
        if (a[i] > m) m = a[i];
    }
    return m;
}

int min(int *a, int n) {
    int m = a[0];
    for (int i = 1; i < n; i++) {
        if (a[i] < m) m = a[i];
    }
    return m;
}

double mean(int *a, int n) {
    double sum = 0;
    for (int i = 0; i < n; i++) {
        sum += a[i];
    }
    return sum / n;
}

double variance(int *a, int n) {
    double m = mean(a, n);
    double sum = 0;
    for (int i = 0; i < n; i++) {
        double diff = a[i] - m;
        sum += diff * diff;
    }
    return sum / n;
}

int main(void) {
    int a[100];
    int n;
    scanf("%d", &n);
    input(a, n);

    output(a, n);
    printf("\n");
    printf("%d\n", max(a, n));
    printf("%d\n", min(a, n));
    printf("%.6f\n", mean(a, n));
    printf("%.6f\n", variance(a, n));

    return 0;
}
```

### Разбор: дисперсия

```
Дисперсия = (1/n) × Σ(xi - mean)²
```

1. Находим среднее `mean`
2. Для каждого элемента вычисляем `(xi - mean)²`
3. Суммируем и делим на n

### Компиляция

```bash
gcc -Wall -Werror -Wextra -o stat src/stat.c
echo "5 1 2 3 4 5" | ./stat
# Вывод:
# 1 2 3 4 5
# 5
# 1
# 3.000000
# 2.000000
```

## Quest 4. search.c

### Задание

Реализовать поиск: найти первый чётный элемент, не меньше среднего, в пределах 3σ, не равный 0.

### Полный код

```c
// src/search.c

#include <math.h>
#include <stdio.h>

void input(int *a, int n) {
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
}

double mean(int *a, int n) {
    double sum = 0;
    for (int i = 0; i < n; i++) sum += a[i];
    return sum / n;
}

double std_dev(int *a, int n) {
    double m = mean(a, n);
    double sum = 0;
    for (int i = 0; i < n; i++) {
        double d = a[i] - m;
        sum += d * d;
    }
    return sqrt(sum / n);
}

int search(int *a, int n) {
    double m = mean(a, n);
    double sd = std_dev(a, n);
    double lower = m - 3 * sd;
    double upper = m + 3 * sd;

    for (int i = 0; i < n; i++) {
        if (a[i] % 2 == 0 && a[i] >= m && a[i] >= lower && a[i] <= upper && a[i] != 0) {
            return a[i];
        }
    }
    return 0;
}

int main(void) {
    int a[100];
    int n;
    scanf("%d", &n);
    input(a, n);

    int result = search(a, n);
    printf("%d\n", result);
    return 0;
}
```

### Разбор: 3-сигма правило

- `mean` — среднее значение
- `std_dev` — стандартное отклонение (σ)
- `m - 3*sd` до `m + 3*sd` — диапазон, в который попадают 99.7% данных

### Компиляция

```bash
gcc -Wall -Werror -Wextra -o search src/search.c -lm
echo "6 1 2 4 6 8 10" | ./search
# Вывод: 4
```

## Quest 5. sort.c

### Задание

Отсортировать 10 целых чисел без `stdlib.h`, используя только указатели.

### Полный код (сортировка выбором)

```c
// src/sort.c

#include <stdio.h>

void input(int *a, int n) {
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
}

void sort(int *a, int n) {
    for (int i = 0; i < n - 1; i++) {
        int min_idx = i;
        for (int j = i + 1; j < n; j++) {
            if (*(a + j) < *(a + min_idx)) {
                min_idx = j;
            }
        }
        // Обмен
        int temp = *(a + i);
        *(a + i) = *(a + min_idx);
        *(a + min_idx) = temp;
    }
}

void output(int *a, int n) {
    for (int i = 0; i < n; i++) {
        if (i > 0) printf(" ");
        printf("%d", a[i]);
    }
}

int main(void) {
    int a[10];
    input(a, 10);
    sort(a, 10);
    output(a, 10);
    return 0;
}
```

### Разбор: сортировка выбором

```
Исходный: [5, 3, 1, 4, 2]

Итерация 1: ищем минимум [3,1,4,2] → 1 → меняем с [5] → [1, 3, 5, 4, 2]
Итерация 2: ищем минимум [3,5,4,2] → 2 → меняем с [3] → [1, 2, 5, 4, 3]
Итерация 3: ищем минимум [5,4,3]   → 3 → меняем с [5] → [1, 2, 3, 4, 5]
Итерация 4: [4,5] → 4 → [1, 2, 3, 4, 5]
```

### Компиляция

# Вывод: -1 0 1 2 2 3 4 7 9 100

## Quest 7. cycle_shift.c

### Задание

Реализовать циклический сдвиг массива влево на k позиций. Если k отрицательное — сдвиг вправо.

### Полный код

```c
// src/cycle_shift.c

#include <stdio.h>

void input(int *a, int n) {
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
}

void cycle_shift(int *a, int n, int k) {
    k = k % n;
    if (k < 0) k += n;

    for (int i = 0; i < k; i++) {
        int last = *(a + n - 1);
        for (int j = n - 1; j > 0; j--) {
            *(a + j) = *(a + j - 1);
        }
        *a = last;
    }
}

void output(int *a, int n) {
    for (int i = 0; i < n; i++) {
        if (i > 0) printf(" ");
        printf("%d", a[i]);
    }
}

int main(void) {
    int a[100];
    int n, k;

    scanf("%d", &n);
    input(a, n);
    scanf("%d", &k);

    cycle_shift(a, n, k);
    output(a, n);

    return 0;
}
```

### Разбор: циклический сдвиг

```
Массив: [1, 2, 3, 4, 5], k = 2

Сдвиг 1: [5, 1, 2, 3, 4]  (последний → первый)
Сдвиг 2: [4, 5, 1, 2, 3]  (ещё раз)

Результат: [4, 5, 1, 2, 3]
```

### Компиляция

## Quest 8. key9part2.c (Bonus)

### Задание

Арифметика больших чисел: сложение и вычитание чисел, представленных как массивы цифр.

### Полный код

```c
// src/key9part2.c

#include <stdio.h>

void input(int *a, int n) {
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
}

void output(int *a, int n) {
    int start = 0;
    while (start < n - 1 && a[start] == 0) start++;
    for (int i = start; i < n; i++) {
        printf("%d", a[i]);
    }
}

void add(int *a, int *b, int *result, int n) {
    int carry = 0;
    for (int i = n - 1; i >= 0; i--) {
        int sum = a[i] + b[i] + carry;
        result[i] = sum % 10;
        carry = sum / 10;
    }
}

void subtract(int *a, int *b, int *result, int n) {
    int borrow = 0;
    for (int i = n - 1; i >= 0; i--) {
        int diff = a[i] - b[i] - borrow;
        if (diff < 0) {
            diff += 10;
            borrow = 1;
        } else {
            borrow = 0;
        }
        result[i] = diff;
    }
}

int main(void) {
    int a[100], b[100], result[100];
    int n;
    char op;

    scanf("%d", &n);
    input(a, n);
    scanf(" %c", &op);
    input(b, n);

    if (op == '+') {
        add(a, b, result, n);
    } else if (op == '-') {
        subtract(a, b, result, n);
    }

    output(result, n);
    return 0;
}
```

### Разбор: сложение столбиком

```
  1 2 3
+ 4 5 6
-------
  5 7 9

Алгоритм (справа налево):
3+6=9, carry=0 → 9
2+5=7, carry=0 → 7
1+4=5, carry=0 → 5
```

### Разбор: вычитание столбиком

```
  5 7 9
- 1 2 3
-------
  4 5 6

Алгоритм (справа налево):
9-3=6, borrow=0 → 6
7-2=5, borrow=0 → 5
5-1=4, borrow=0 → 4
```

## Шпаргалка: указатели

### Основы

```c
int x = 42;      // Переменная
int *p = &x;     // p — указатель на x (хранит адрес x)
*p = 100;        // Изменяем x через указатель: x теперь 100

printf("%d", x);    // 100
printf("%d", *p);   // 100
printf("%p", p);    // адрес x в hex
```

### Указатели и массивы

```c
int a[5] = {1, 2, 3, 4, 5};
int *p = a;         // p указывает на a[0]

a[0] == *p          // 1
a[1] == *(p+1)      // 2
a[2] == *(p+2)      // 3

// Индексация — это сахар:
a[i]  ←→  *(a+i)  ←→  *(p+i)
```

### Передача массива в функцию

```c
void func(int *a, int n);   // Массив передаётся как указатель
func(my_array, 10);         // Вызов

// Внутри функции:
// a[0] — первый элемент
// a[1] — второй
// a[i] — i-й
```
