## Основная часть: pong.c

### Правила игры

- Поле 80×25 символов
- Две ракетки (высота 3 символа) — слева (A/Z) и справа (K/M)
- Мяч — 1 символ
- Счёт отображается сверху
- Игра до 21 очка
- Пошаговый режим: каждый ход — ввод одного символа, потом вывод поля
- Пробел — пропустить ход

### Полный код src/pong.c

```c
// src/pong.c

#include <stdio.h>

// Размеры поля
#define WIDTH 80
#define HEIGHT 25

// Начальные позиции
#define PADDLE_SIZE 3
#define WIN_SCORE 21

int main(void) {
    // Позиция ракеток (верхний край)
    int left_paddle = 10;
    int right_paddle = 10;

    // Позиция мяча
    int ball_x = 40;
    int ball_y = 12;

    // Направление мяча
    int ball_dx = 1;
    int ball_dy = 1;

    // Счёт
    int score_left = 0;
    int score_right = 0;

    // Флаг окончания игры
    int game_over = 0;

    printf("Welcome to Pong!\n");
    printf("Player 1: A (up) / Z (down)\n");
    printf("Player 2: K (up) / M (down)\n");
    printf("Space to skip turn\n");
    printf("First to %d wins!\n\n", WIN_SCORE);

    while (!game_over) {
        // === ВЫВОД ПОЛЯ ===

        // Верхняя граница + счёт
        printf("\r");
        for (int i = 0; i < WIDTH; i++) printf("-");
        printf("  Score: %d - %d\n", score_left, score_right);

        // Игровое поле
        for (int y = 1; y < HEIGHT - 1; y++) {
            printf("|");  // Левая граница

            for (int x = 0; x < WIDTH - 2; x++) {
                // Левая ракетка
                if (x == 2 && y >= left_paddle && y < left_paddle + PADDLE_SIZE) {
                    printf("|");
                }
                // Правая ракетка
                else if (x == WIDTH - 5 && y >= right_paddle && y < right_paddle + PADDLE_SIZE) {
                    printf("|");
                }
                // Мяч
                else if (x == ball_x - 1 && y == ball_y) {
                    printf("o");
                }
                // Пустое пространство
                else {
                    printf(" ");
                }
            }

            printf("|\n");  // Правая граница
        }

        // Нижняя граница
        for (int i = 0; i < WIDTH; i++) printf("-");
        printf("\n");

        // === ПРОВЕРКА ПОБЕДЫ ===
        if (score_left >= WIN_SCORE || score_right >= WIN_SCORE) {
            game_over = 1;
            if (score_left >= WIN_SCORE)
                printf("\nPlayer 1 wins! %d - %d\n", score_left, score_right);
            else
                printf("\nPlayer 2 wins! %d - %d\n", score_left, score_right);
            break;
        }

        // === ВВОД ===
        printf("Enter move (A/Z/K/M or space): ");
        char input;
        scanf(" %c", &input);

        // === ОБРАБОТКА ВВОДА ===
        if (input == 'a' || input == 'A') {
            if (left_paddle > 1) left_paddle--;
        } else if (input == 'z' || input == 'Z') {
            if (left_paddle < HEIGHT - PADDLE_SIZE - 1) left_paddle++;
        } else if (input == 'k' || input == 'K') {
            if (right_paddle > 1) right_paddle--;
        } else if (input == 'm' || input == 'M') {
            if (right_paddle < HEIGHT - PADDLE_SIZE - 1) right_paddle++;
        }

        // === ДВИЖЕНИЕ МЯЧА ===
        ball_x += ball_dx;
        ball_y += ball_dy;

        // Отскок от верхней/нижней стенки
        if (ball_y <= 1 || ball_y >= HEIGHT - 2) {
            ball_dy = -ball_dy;
        }

        // Отскок от ракеток
        if (ball_x == 4 && ball_y >= left_paddle && ball_y < left_paddle + PADDLE_SIZE) {
            ball_dx = -ball_dx;
        }
        if (ball_x == WIDTH - 6 && ball_y >= right_paddle && ball_y < right_paddle + PADDLE_SIZE) {
            ball_dx = -ball_dx;
        }

        // Гол
        if (ball_x <= 0) {
            score_right++;
            ball_x = 40;
            ball_y = 12;
            ball_dx = 1;
        }
        if (ball_x >= WIDTH - 1) {
            score_left++;
            ball_x = 40;
            ball_y = 12;
            ball_dx = -1;
        }
    }

    return 0;
}
```

### Как это работает

#### Отрисовка поля

Поле рисуется построчно с помощью `printf`:

```
+------------------------------------------------------------------------+
|  Score: 3 - 2                                                          |
|                                                                        |
|   |                              o                                     |
|   |                                                              |      |
|   |                                                              |      |
|   |                                                              |      |
|                                                                        |
+------------------------------------------------------------------------+
```

- Каждый `printf` выводит один символ или пробел
- Внутренний цикл `for (x = 0; x < WIDTH-2; x++)` рисует строку
- Внешний цикл `for (y = 1; y < HEIGHT-1; y++)` двигается по строкам

#### Движение мяча

```
ball_x += ball_dx;  //_dx = 1 (вправо) или -1 (влево)
ball_y += ball_dy;  // dy = 1 (вниз) или -1 (вверх)
```

При столкновении со стенкой или ракеткой `dx` или `dy` меняют знак.

#### Обработка ввода

```
scanf(" %c", &input);  // Пробел перед %c пропускает Enter и пробелы
```

| Символ | Действие |
|--------|----------|
| A/a | Ракетка 1 вверх |
| Z/z | Ракетка 1 вниз |
| K/k | Ракетка 2 вверх |
| M/m | Ракетка 2 вниз |
| Пробел | Пропустить ход |

### Компиляция и запуск

```bash
gcc -Wall -Werror -Wextra -o pong src/pong.c
./pong
```

## Bonus: pong_interactive.c (ncurses)

### Что такое ncurses

ncurses — библиотека для создания интерфейсов в терминале. Позволяет:
- Ставить символ в любую точку экрана
- Не мерцает при перерисовке
- Обрабатывать нажатия клавиш без Enter

### Библиотека ncurses

Для бонусной части необходима библиотека `ncurses`. Если она не установлена — обратитесь к администратору.

### Полный код src/pong_interactive.c

```c
// src/pong_interactive.c

#include <ncurses.h>
#include <unistd.h>

#define WIDTH 80
#define HEIGHT 25
#define PADDLE_SIZE 3
#define WIN_SCORE 21
#define DELAY 100000  // Начальная задержка (микросекунды)

int main(void) {
    // Инициализация ncurses
    initscr();
    cbreak();
    noecho();
    curs_set(0);          // Скрыть курсор
    keypad(stdscr, TRUE); // Включить спец. клавиши
    nodelay(stdscr, TRUE); // getch() не блокирует

    // Позиции
    int left_paddle = 10, right_paddle = 10;
    int ball_x = 40, ball_y = 12;
    int ball_dx = 1, ball_dy = 1;
    int score_left = 0, score_right = 0;
    int speed = DELAY;
    int ch;

    while (score_left < WIN_SCORE && score_right < WIN_SCORE) {
        erase();  // Очистить экран

        // Верхняя граница + счёт
        mvprintw(0, 0, "Score: %d - %d  |  A/Z: P1  K/M: P2  Space: quit",
                 score_left, score_right);

        // Рисуем поле
        for (int i = 0; i < WIDTH; i++) {
            mvaddch(1, i, '-');
            mvaddch(HEIGHT - 1, i, '-');
        }

        // Левая ракетка
        for (int i = 0; i < PADDLE_SIZE; i++) {
            mvaddch(2 + left_paddle + i, 3, '|');
        }

        // Правая ракетка
        for (int i = 0; i < PADDLE_SIZE; i++) {
            mvaddch(2 + right_paddle + i, WIDTH - 4, '|');
        }

        // Мяч
        mvaddch(2 + ball_y, ball_x, 'o');

        refresh();  // Обновить экран

        // Ввод (неблокирующий)
        ch = getch();
        if (ch == 'a' && left_paddle > 0) left_paddle--;
        if (ch == 'z' && left_paddle < HEIGHT - PADDLE_SIZE - 3) left_paddle++;
        if (ch == 'k' && right_paddle > 0) right_paddle--;
        if (ch == 'm' && right_paddle < HEIGHT - PADDLE_SIZE - 3) right_paddle++;
        if (ch == ' ') break;

        // Управление скоростью
        if (ch == 'A') {         // Стрелка вверх — ускорить (уменьшить задержку)
            if (speed > 20000) speed -= 10000;
        }
        if (ch == 'B') {         // Стрелка вниз — замедлить
            if (speed < 200000) speed += 10000;
        }

        // Движение мяча
        ball_x += ball_dx;
        ball_y += ball_dy;

        if (ball_y <= 0 || ball_y >= HEIGHT - 3) ball_dy = -ball_dy;

        // Отскок от ракеток
        if (ball_x == 4 && ball_y >= left_paddle && ball_y < left_paddle + PADDLE_SIZE)
            ball_dx = -ball_dx;
        if (ball_x == WIDTH - 5 && ball_y >= right_paddle && ball_y < right_paddle + PADDLE_SIZE)
            ball_dx = -ball_dx;

        // Гол
        if (ball_x <= 0) { score_right++; ball_x = 40; ball_y = 12; ball_dx = 1; }
        if (ball_x >= WIDTH - 1) { score_left++; ball_x = 40; ball_y = 12; ball_dx = -1; }

        usleep(speed);  // Задержка (скорость игры)
    }

    // Вывод победителя
    endwin();
    if (score_left >= WIN_SCORE)
        printf("Player 1 wins! %d - %d\n", score_left, score_right);
    else
        printf("Player 2 wins! %d - %d\n", score_left, score_right);

    return 0;
}
```

### Разбор ключевых функций ncurses

| Функция | Что делает |
|---------|-----------|
| `initscr()` | Инициализация режима curses |
| `endwin()` | Завершение режима curses |
| `erase()` | Очистить экран (лучше чем clear для анимации) |
| `refresh()` | Обновить реальный экран |
| `mvaddch(y, x, ch)` | Поставить символ в позицию (y, x) |
| `mvprintw(y, x, fmt, ...)` | Напечатать текст в позиции |
| `getch()` | Прочитать нажатую клавишу (неблокирующую) |
| `usleep(microseconds)` | Задержка в микросекундах |
| `noecho()` | Не отображать вводимые символы |
| `curs_set(0)` | Скрыть курсор |
| `nodelay(stdscr, TRUE)` | getch() не ждёт нажатия |

### Горячие клавиши ncurses

| Клавиша | Код getch() | Действие |
|---------|-------------|----------|
| A | 'a' | Paddle 1 вверх |
| Z | 'z' | Paddle 1 вниз |
| K | 'k' | Paddle 2 вверх |
| M | 'm' | Paddle 2 вниз |
| Стрелка вверх | KEY_UP | Ускорить |
| Стрелка вниз | KEY_DOWN | Замедлить |
| Пробел | ' ' | Выход |

### Компиляция с ncurses

```bash
gcc -Wall -Werror -Wextra -o pong_interactive src/pong_interactive.c -lncurses
./pong_interactive
```

### Важные замечания

1. **`-lncurses`** — флаг линковки с библиотекой ncurses
2. **`usleep()`** — регулирует скорость игры. Меньше = быстрее
3. **`nodelay`** — без этого `getch()` блокирует программу до нажатия клавиши
4. **`erase()` вместо `clear()`** — erase не мерцает

## Шпаргалка: ncurses

### Основные функции

```c
#include <ncurses.h>

initscr();           // Начать
endwin();            // Закончить
clear();             // Очистить экран
refresh();           // Обновить

// Вывод
printw("text");     // Как printf
mvprintw(y, x, "text");  // Текст в позиции
addch('c');          // Символ в текущей позиции
mvaddch(y, x, 'c'); // Символ в позиции

// Ввод
ch = getch();        // Прочитать клавишу
nodelay(stdscr, TRUE);  // Неблокирующий ввод

// Настройка
noecho();            // Не отображать ввод
cbreak();            // Поимвольный ввод
curs_set(0);         // Скрыть курсор
keypad(stdscr, TRUE); // Спец. клавиши (стрелки и т.д.)
```

### Задержка

```c
#include <unistd.h>

usleep(100000);  // 100000 микросекунд = 0.1 секунды
```
